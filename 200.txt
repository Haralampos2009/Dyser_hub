-- Helper functions
local function scaledColor(color, scale)
	return Color3.new(
		math.clamp(color.R * scale, 0, 1),
		math.clamp(color.G * scale, 0, 1),
		math.clamp(color.B * scale, 0, 1)
	)
end

local function trackAccentBackground(object)
	table.insert(accentBackgrounds, object)
	object.BackgroundColor3 = COLORS.red
	return object
end

local function trackAccentDarkBackground(object)
	table.insert(accentDarkBackgrounds, object)
	object.BackgroundColor3 = COLORS.redDark
	return object
end

local function trackAccentStroke(stroke)
	table.insert(accentStrokes, stroke)
	stroke.Color = COLORS.red
	return stroke
end

local function trackAccentGradient(gradient)
	table.insert(accentGradients, gradient)
	gradient.Color = ColorSequence.new(COLORS.redDark, COLORS.red)
	return gradient
end

function Library.SetAccentColor(color)
	COLORS.red = color
	COLORS.redDark = scaledColor(color, 0.55)

	for _, object in ipairs(accentBackgrounds) do
		if object and object.Parent then
			object.BackgroundColor3 = COLORS.red
		end
	end

	for _, object in ipairs(accentDarkBackgrounds) do
		if object and object.Parent then
			object.BackgroundColor3 = COLORS.redDark
		end
	end

	for _, stroke in ipairs(accentStrokes) do
		if stroke and stroke.Parent then
			stroke.Color = COLORS.red
		end
	end

	for _, gradient in ipairs(accentGradients) do
		if gradient and gradient.Parent then
			gradient.Color = ColorSequence.new(COLORS.redDark, COLORS.red)
		end
	end

	if selectedTopLeftTab and selectedTopLeftTab.Parent then
		selectedTopLeftTab.BackgroundColor3 = COLORS.red
	end

	if selectedSideButtonGradient and selectedSideButtonGradient.Parent then
		selectedSideButtonGradient.Color = ColorSequence.new(COLORS.redDark, COLORS.red)
	end
end

function Library.AddCorner(parent, radius)
	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, radius)
	corner.Parent = parent
	return corner
end

function Library.AddStroke(parent, color, thickness, transparency)
	local stroke = Instance.new("UIStroke")
	stroke.Color = color
	stroke.Thickness = thickness or 1
	stroke.Transparency = transparency or 0
	stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	stroke.Parent = parent
	return stroke
end

function Library.AddGradient(parent, colorA, colorB, rotation)
	local gradient = Instance.new("UIGradient")
	gradient.Color = ColorSequence.new(colorA, colorB)
	gradient.Rotation = rotation or 90
	gradient.Parent = parent
	return gradient
end

function Library.MakeText(className, props)
	local object = Instance.new(className)
	object.BackgroundTransparency = props.BackgroundTransparency or 1
	object.Text = props.Text or ""
	object.TextColor3 = props.TextColor3 or COLORS.text
	object.TextSize = props.TextSize or 14
	object.TextXAlignment = props.TextXAlignment or Enum.TextXAlignment.Left
	object.TextYAlignment = props.TextYAlignment or Enum.TextYAlignment.Center
	object.Font = props.Font or Enum.Font.Code
	object.BorderSizePixel = 0

	for key, value in pairs(props) do
		if object[key] ~= nil then
			object[key] = value
		end
	end

	return object
end

function Library.RegisterSearchable(name, sideTab, element)
	table.insert(searchableElements, {
		name = name,
		sideTab = sideTab,
		element = element
	})
end

-- Main window creation
function Library.New(config)
	if config then
		for key, value in pairs(config) do
			if Library.Config[key] then
				Library.Config[key] = value
			end
		end
	end
	
	COLORS = Library.Config.COLORS
	
	-- Remove old GUI
	local oldGui = playerGui:FindFirstChild("NumberedMenuReplica")
	if oldGui then
		oldGui:Destroy()
	end

	local screenGui = Instance.new("ScreenGui")
	screenGui.Name = "NumberedMenuReplica"
	screenGui.IgnoreGuiInset = true
	screenGui.ResetOnSpawn = false
	screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
	screenGui.Parent = playerGui
	screenGuiRef = screenGui

	local dim = Instance.new("Frame")
	dim.Name = "SoftGameDim"
	dim.Size = UDim2.fromScale(1, 1)
	dim.BackgroundColor3 = Color3.fromRGB(8, 14, 11)
	dim.BackgroundTransparency = Library.Config.DIM_TRANSPARENCY
	dim.BorderSizePixel = 0
	dim.Parent = screenGui

	local window = Instance.new("Frame")
	window.Name = "MenuWindow"
	window.Position = UDim2.new(Library.Config.MENU.position.x, 0, Library.Config.MENU.position.y, 0)
	window.Size = UDim2.new(Library.Config.MENU.size.x, 0, Library.Config.MENU.size.y, 0)
	window.BackgroundColor3 = COLORS.black
	window.BorderSizePixel = 0
	window.Visible = false
	window.Parent = screenGui
	Library.AddStroke(window, COLORS.red, 1, 0.35)
	windowRef = window

	local windowScale = Instance.new("UIScale")
	windowScale.Scale = Library.Config.MENU.scale
	windowScale.Parent = window

	local sizeLimit = Instance.new("UISizeConstraint")
	sizeLimit.MinSize = Vector2.new(Library.Config.MENU.minSize.x, Library.Config.MENU.minSize.y)
	sizeLimit.MaxSize = Vector2.new(Library.Config.MENU.maxSize.x, Library.Config.MENU.maxSize.y)
	sizeLimit.Parent = window

	-- Build UI structure
	Library.BuildTopBar(window)
	Library.BuildBody(window)
	Library.BuildColorPicker(window)
	
	-- Setup toggle key
	UserInputService.InputBegan:Connect(function(input, gameProcessed)
		if gameProcessed then return end
		if input.KeyCode == Library.Config.TOGGLE_KEY then
			Library.ToggleMenu()
		end
	end)

	-- Auto open menu
	Library.OpenMenu()

	return {
		window = window,
		screenGui = screenGui,
		dim = dim,
	}
end

		for _, otherTab in ipairs(topLeftTabButtons) do
			otherTab.button.BackgroundColor3 = Color3.fromRGB(8, 8, 9)

		end

		for name, content in pairs(topTabContents) do
			content.Visible = (name == tabName)
		end
		
		local sideTabsForTop = TOP_TAB_SIDE_TABS[tabName] or {}
		for _, entry in ipairs(sideButtons) do
			local shouldShow = false
			for _, tabNameInList in ipairs(sideTabsForTop) do
				if entry.name == tabNameInList then
					shouldShow = true
					break
				end
			end
			entry.button.Visible = shouldShow
			entry.page.Visible = false
		end
		
		for _, entry in ipairs(sideButtons) do
			if entry.button.Visible then
				Library.SelectSideButton(entry.button, entry.gradient, entry.name, entry.page)
				break
			end
		end
	end

	end

-- Select side button
function Library.SelectSideButton(button, buttonGradient, tabName, page)
	for _, entry in ipairs(sideButtons) do
		entry.button.TextColor3 = COLORS.white
		entry.button.TextStrokeColor3 = COLORS.white
		entry.button.BackgroundColor3 = Color3.fromRGB(25, 25, 28)
		entry.gradient.Enabled = false
	end

	button.BackgroundColor3 = COLORS.red
	buttonGradient.Enabled = true
	buttonGradient.Color = ColorSequence.new(COLORS.redDark, COLORS.red)
	selectedSideButtonGradient = buttonGradient
	button.TextColor3 = COLORS.white
	button.TextStrokeColor3 = COLORS.white

	-- Update tab text
	if windowRef then
		local tabText = windowRef.TopBar.TopTab2
		if tabText then
			tabText.Text = tabName
		end
	end

	for _, otherPage in pairs(tabPages) do
		otherPage.Visible = false
	end

	if page then
		page.Visible = true
		activeTabPage = page
	end
end

-- Build search box
function Library.BuildSearchBox(rightTopLine)
	local searchBox = Instance.new("Frame")
	searchBox.Name = "SearchBox"
	searchBox.Position = UDim2.fromOffset(5, 24)
	searchBox.Size = UDim2.new(1, -10, 0, 22)
	searchBox.BackgroundColor3 = Color3.fromRGB(8, 8, 9)
	searchBox.BorderSizePixel = 0
	searchBox.Parent = rightTopLine
	Library.AddCorner(searchBox, 4)
	Library.AddStroke(searchBox, COLORS.lineLight, 1, 0.15)

	local searchInput = Instance.new("TextBox")
	searchInput.Name = "SearchInput"
	searchInput.Size = UDim2.new(1, -30, 1, 0)
	searchInput.Position = UDim2.fromOffset(10, 0)
	searchInput.BackgroundTransparency = 1
	searchInput.Text = ""
	searchInput.PlaceholderText = "Search..."
	searchInput.PlaceholderColor3 = COLORS.muted
	searchInput.TextColor3 = COLORS.text
	searchInput.TextSize = 12
	searchInput.Font = Enum.Font.Code
	searchInput.ClearTextOnFocus = false
	searchInput.Parent = searchBox

	searchInput.FocusLost:Connect(function(enterPressed)
		if not enterPressed then return end
		
		local searchTerm = string.lower(searchInput.Text)
		if searchTerm == "" then return end
		
		for _, element in ipairs(searchableElements) do
			if string.lower(element.name) == searchTerm or string.find(string.lower(element.name), searchTerm, 1, true) then
				for _, sideEntry in ipairs(sideButtons) do
					if sideEntry.name == element.sideTab then
						Library.SelectSideButton(sideEntry.button, sideEntry.gradient, sideEntry.name, sideEntry.page)
						break
					end
				end
				break
			end
		end
		
		searchInput.Text = ""
	end)
end

function Library.AddToggle(tabName, toggleName, callback)
	local page = tabPages[tabName]
	if not page then return nil end
	
	local row = Instance.new("Frame")
	row.Size = UDim2.new(1, 0, 0, Library.Config.SIZES.rowHeight)
	row.BackgroundColor3 = COLORS.black
	row.BackgroundTransparency = 0.04
	row.BorderSizePixel = 0
	row.Parent = page
	
	local label = Library.MakeText("TextLabel", {
		Name = "Label",
		Size = UDim2.new(1, -30, 1, 0),
		Text = toggleName,
		TextSize = Library.Config.TEXT_SIZES.label,
		TextColor3 = Color3.fromRGB(205, 205, 210),
	})
	label.Parent = row
	
	Library.RegisterSearchable(toggleName, tabName, row)
	
	local padding = Instance.new("UIPadding")
	padding.PaddingLeft = UDim.new(0, 3)
	padding.Parent = label
	
	local enabled = false
	
	local box = Library.MakeText("TextButton", {
		Name = "ToggleX",
		AnchorPoint = Vector2.new(1, 0.5),
		Position = UDim2.new(1, -3, 0.5, 0),
		Size = UDim2.fromOffset(Library.Config.SIZES.toggleBoxSize, Library.Config.SIZES.toggleBoxSize),
		BackgroundTransparency = 0,
		BackgroundColor3 = Color3.fromRGB(16, 10, 15),
		Text = "x",
		TextColor3 = COLORS.pink,
		TextSize = 18,
		Font = Library.Config.FONTS.bold,
		TextXAlignment = Enum.TextXAlignment.Center,
		AutoButtonColor = false,
	})
	box.Parent = row
	Library.AddCorner(box, 4)
	Library.AddStroke(box, Color3.fromRGB(48, 28, 43), 1, 0.05)
	
	local function setEnabled(value)
		enabled = value
		box.Text = enabled and "✓" or "x"
		box.TextColor3 = enabled and Color3.fromRGB(70, 230, 120) or COLORS.pink
		box.BackgroundColor3 = enabled and Color3.fromRGB(9, 30, 15) or Color3.fromRGB(16, 10, 15)
		row.BackgroundTransparency = enabled and 0.02 or 0.14
		
		if callback then
			callback(enabled)
		end
	end
	
	setEnabled(false)
	
	box.MouseButton1Click:Connect(function()
		setEnabled(not enabled)
	end)
	
	row.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			setEnabled(not enabled)
		end
	end)
	
	return {
		setState = setEnabled,
		getState = function() return enabled end
	}
end
